announce integer func sum(integer x, integer y)
{
	announce integer tmp = x + y;
	return tmp;
}

announce integer func fact(integer n)
{
	announce integer acc;
	acc = 1;

	cycle (n)
	{
	     acc = acc * 2;
	}
	return acc;
}

announce boolean func bool(integer w1, integer w2)
{
	announce boolean res;
	res = w2 > w1;
	return res;
}
announce integer func test_fi(integer w1, integer w2)
{
	announce integer res = w1 + power_of(w2, 3);

	return res;
}
